Compare MAD Sheets Function

function compareMAD(setABeginning, setA, setBBeginning, setB) {
  if(setABeginning == "" || setBBeginning == "") {
    var result = "Please enter data in both sets.";
  } else {
    if(setA > setB) {
      var result = "Set B has less variability.";
    } else if(setA < setB) {
      var result = "Set A has less variability.";
    } else if(setA == setB) {
      var result = "Sets A and B have the same variability.";
    } else {
      var result = "Oops! Something went wrong.";
    }
  }
  return result;
}
